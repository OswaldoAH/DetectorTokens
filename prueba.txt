class nodo
{
	private int valor;
	private nodo siguiente;
	
	public nodo()
	{
		valor=0;
		siguiente = null;
	}
	
	public void setValor(int n)
	{
		valor=n;
	}
	
	public int getValor()
	{
		return valor;
	}
	
	public void setSiguiente(nodo n)
	{
		siguiente = n;
	}
	
	public nodo getSiguiente()
	{
		return siguiente;
	}
}
